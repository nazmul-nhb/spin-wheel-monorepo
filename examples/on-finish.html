2. onFinish Callback

Pass a callback in the widget config. It fires with the SpinResult after each spin — handy for analytics, toasts, or decoupled logic.

import { SpinWheelWidget } from '@spin-wheel/widget';

const widget = SpinWheelWidget.create('#wheel', {
  segments,
  // ✅ onFinish fires after each spin
  onFinish(result) {
    console.log('Winner:', result.segment.label);
    console.log('Index:', result.index);
    console.log('Data:', result.segment.data);
  },
  // ✅ onStateChange tracks lifecycle
  onStateChange(state) {
    console.log('State:', state);
    // 'spinning' → 'finished' → 'idle'
  },
});

// Just trigger the spin — no need to await for the result
widget.spin();
