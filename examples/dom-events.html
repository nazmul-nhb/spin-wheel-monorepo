// Import types for DOM event augmentation
import '@spin-wheel/embed';
// or: import type {} from '@spin-wheel/embed';

// ✅ Fully typed — e.detail is SpinResult
document.addEventListener('spinwheel:finish', (e) => {
  console.log(e.detail.index);
  console.log(e.detail.segment.label);
  console.log(e.detail.finalAngle);
});

// ✅ Typed — e.detail.state is WheelState
document.addEventListener('spinwheel:state', (e) => {
  console.log(e.detail.state);
});

// ✅ Typed — e.detail.widget is SpinWheelWidget
document.addEventListener('spinwheel:ready', (e) => {
  const widget = e.detail.widget;
  widget.spin();
});
