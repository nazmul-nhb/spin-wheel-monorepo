Source Code

import { WheelEngine } from '@spin-wheel/core';

const engine = new WheelEngine({
  segments: [
    { id: 'pizza', label: '🍕 Pizza', weight: 2 },
    { id: 'tacos', label: '🌮 Tacos', weight: 3 },
    // ...
  ],
  seed: 'my-seed', // deterministic!
});

// ✅ Synchronous — result is known instantly
const result = engine.spin();
console.log(result.segment.label);

// Retrieve the last result again later
const same = engine.getLastResult();

// Reset and spin again
engine.reset();
const next = engine.spin();

// Check engine state
engine.getState();     // 'idle' | 'spinning' | 'finished'
engine.getSegments();  // readonly frozen segments
